######	Learn GIT & GITHUB	######

1.	git --version
	// This command is used for checking the git version
	
2. 	git update-git-for-windows
	// This command is used for checking the git update
	
3.	git config --global user.name "USERNAME"
	// This command is used for updating the username
	
4.	git config --global user.email "EMAIL-ADDRESS"
	// This command is used for updating the email
	
5.	git config --global --edit
	// This command is used for updating the email and the username
	
6.	mkdir DIRECTORY-NAME
	// This command is used for creting a new directory
	
7.	cd DIRECTORY-NAME
	// This command is used to go inside the directory
	
8.	git status
	// This command is used to check the status for the tracked and untracked files in git
	
9.	git add FULL-FILE-NAME
	// This command is used to add a specific file in staging area before commit
	
10.	git add -all or git add .
	// This command is used to add all the untracked files in staging area

11.	git commit -m "MESSAGE YOU WANT TO SPECIFIED"
	// This command is used to commit the files available in staging area

12.	git branch
	// This command is used to check all the available branches in the repo

13.	git branch BRANCH-NAME
	// This command is used to crate a new branch in same repository

14.	git checkout BRANCH-NAME
	// This command is used to checkout in the desired branch.
	
15.	git merge BRANCH-NAME
	//This commit is used to merge the BRANCH_NAME banch into the existing branch

16.	.gitignore
	// This file is used to ignore the files which we dont want commit or put into staging

17.	git clone REPO-URL
	// This command is used to clone a repo available on github or any git based platform

18.	git push
	// This command is used to push the changes from local git to the remote git
